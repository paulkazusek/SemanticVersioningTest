# Semantic Versioning with Git

git commit -m "first commit"

git tag -a v0.0.0 -m "version 0.0.0"

git push -u origin master

git push -u origin master --tags


add new feature

git commit -m "first feature"

git tag -a v0.1.0 -m "version 0.1.0"

git push -u origin master

git push -u origin master --tags
